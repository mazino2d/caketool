 16class PermutationExplainer(ModelExplainer, BaseEstimator):
 17    """SHAP Permutation-based model explainer compatible with any sklearn model.
 18
 19    Uses ``shap.PermutationExplainer`` under the hood, which is model-agnostic
 20    and works by permuting features to estimate SHAP values. No need to extract
 21    internals from pipelines — pass the full model directly.
 22
 23    Parameters
 24    ----------
 25    model : object
 26        Fitted sklearn-compatible model. Must have ``predict_proba`` or
 27        ``predict`` method.
 28    background_data : array-like, optional
 29        Background dataset used as the reference distribution when computing
 30        SHAP values. If None, a subsample of the fit data is used.
 31    n_background_samples : int, optional
 32        Number of samples to draw from background_data (or fit data when
 33        background_data is None). Defaults to 100.
 34
 35    Attributes
 36    ----------
 37    shap_values_ : np.ndarray of shape (n_samples, n_features)
 38        SHAP values computed after calling fit. For binary classification,
 39        values correspond to the positive class.
 40    feature_names_ : list of str
 41        Feature names inferred from the input data (DataFrame columns or
 42        auto-generated as f0, f1, ...).
 43    explainer_ : shap.PermutationExplainer
 44        Fitted SHAP explainer instance.
 45
 46    Examples
 47    --------
 48    >>> from sklearn.ensemble import GradientBoostingClassifier
 49    >>> from caketool.explainability import PermutationExplainer
 50    >>> model = GradientBoostingClassifier().fit(X_train, y_train)
 51    >>> explainer = PermutationExplainer(model=model)
 52    >>> explainer.fit(X_test)
 53    >>> importance = explainer.get_feature_importance()
 54    """
 55
 56    def __init__(
 57        self,
 58        model,
 59        background_data=None,
 60        n_background_samples: int = 100,
 61    ) -> None:
 62        self.model = model
 63        self.background_data = background_data
 64        self.n_background_samples = n_background_samples
 65        self._is_fitted = False

 67    def _check_fitted(self) -> None:
 68        if not self._is_fitted:
 69            raise RuntimeError(_NOT_FITTED_MSG)

 71    def _get_predict_fn(self):
 72        """Return predict_proba if available, else predict."""
 73        if hasattr(self.model, "predict_proba"):
 74            return self.model.predict_proba
 75        return self.model.predict

 77    @require_dependencies("shap")
 78    def fit(self, X) -> PermutationExplainer:
 79        """Compute SHAP values for the given input data.
 80
 81        Parameters
 82        ----------
 83        X : pd.DataFrame or np.ndarray of shape (n_samples, n_features)
 84            Input data to explain.
 85
 86        Returns
 87        -------
 88        self : PermutationExplainer
 89        """
 90        import shap
 91
 92        if isinstance(X, pd.DataFrame):
 93            self.feature_names_ = list(X.columns)
 94            self._X_fit = X.copy()
 95        else:
 96            self.feature_names_ = [f"f{i}" for i in range(X.shape[1])]
 97            self._X_fit = pd.DataFrame(X, columns=self.feature_names_)
 98
 99        predict_fn = self._get_predict_fn()
100
101        if self.background_data is not None:
102            n = min(self.n_background_samples, len(self.background_data))
103            bg = shap.sample(self.background_data, n)
104        else:
105            n = min(self.n_background_samples, len(self._X_fit))
106            bg = shap.sample(self._X_fit, n)
107
108        self.explainer_ = shap.PermutationExplainer(predict_fn, bg)
109        explanation = self.explainer_(self._X_fit)
110
111        raw_values = explanation.values
112
113        # Normalise binary classification: take positive-class SHAP values
114        if isinstance(raw_values, list):
115            raw_values = raw_values[1] if len(raw_values) == 2 else raw_values[0]
116        elif isinstance(raw_values, np.ndarray) and raw_values.ndim == 3:
117            raw_values = raw_values[:, :, 1]
118
119        self.shap_values_ = np.array(raw_values)
120
121        # Extract base value from first sample (same for all samples)
122        base = np.array(explanation.base_values)
123        if base.ndim == 2:
124            # (n_samples, n_classes) — binary classification: take positive class
125            self.base_value_ = float(base[0, 1]) if base.shape[1] >= 2 else float(base[0, 0])
126        elif base.ndim == 1:
127            self.base_value_ = float(base[0])
128        else:
129            self.base_value_ = float(base)
130
131        self._is_fitted = True
132        return self

134    def get_feature_importance(self) -> pd.DataFrame:
135        """Return global feature importance based on mean absolute SHAP values.
136
137        Returns
138        -------
139        pd.DataFrame
140            Columns: rank, feature, importance_pct, direction, mean_abs_shap.
141
142            - rank: importance ranking (1 = most important)
143            - feature: feature name
144            - importance_pct: normalized importance (0-1 scale, sums to 1)
145            - direction: ``"positive"`` if feature generally increases prediction,
146              ``"negative"`` if it decreases prediction (based on mean SHAP value)
147            - mean_abs_shap: mean absolute SHAP value across all samples
148
149            Sorted by importance descending.
150
151        Raises
152        ------
153        RuntimeError
154            If called before fit.
155        """
156        self._check_fitted()
157        mean_abs = np.abs(self.shap_values_).mean(axis=0)
158        mean_shap = self.shap_values_.mean(axis=0)
159
160        total_importance = mean_abs.sum()
161        importance_pct = mean_abs / total_importance if total_importance > 0 else mean_abs
162
163        df = pd.DataFrame(
164            {
165                "feature": self.feature_names_,
166                "mean_abs_shap": mean_abs,
167                "importance_pct": importance_pct,
168                "direction": ["positive" if v >= 0 else "negative" for v in mean_shap],
169            }
170        )
171        df = df.sort_values("mean_abs_shap", ascending=False).reset_index(drop=True)
172        df["rank"] = np.arange(1, len(df) + 1)
173        return df[["rank", "feature", "importance_pct", "direction", "mean_abs_shap"]]

SHAP Permutation-based model explainer compatible with any sklearn model.

Uses shap.PermutationExplainer under the hood, which is model-agnostic and works by permuting features to estimate SHAP values. No need to extract internals from pipelines — pass the full model directly.

Parameters

model : object Fitted sklearn-compatible model. Must have predict_proba or predict method. background_data : array-like, optional Background dataset used as the reference distribution when computing SHAP values. If None, a subsample of the fit data is used. n_background_samples : int, optional Number of samples to draw from background_data (or fit data when background_data is None). Defaults to 100.

Attributes

shap_values_ : np.ndarray of shape (n_samples, n_features) SHAP values computed after calling fit. For binary classification, values correspond to the positive class. feature_names_ : list of str Feature names inferred from the input data (DataFrame columns or auto-generated as f0, f1, ...). explainer_ : shap.PermutationExplainer Fitted SHAP explainer instance.

>>> from caketool.explainability import PermutationExplainer

>>> explainer = PermutationExplainer(model=model)
>>> explainer.fit(X_test)

 79    def fit(self, X) -> PermutationExplainer:
 80        """Compute SHAP values for the given input data.
 81
 82        Parameters
 83        ----------
 84        X : pd.DataFrame or np.ndarray of shape (n_samples, n_features)
 85            Input data to explain.
 86
 87        Returns
 88        -------
 89        self : PermutationExplainer
 90        """
 91        import shap
 92
 93        if isinstance(X, pd.DataFrame):
 94            self.feature_names_ = list(X.columns)
 95            self._X_fit = X.copy()
 96        else:
 97            self.feature_names_ = [f"f{i}" for i in range(X.shape[1])]
 98            self._X_fit = pd.DataFrame(X, columns=self.feature_names_)
 99
100        predict_fn = self._get_predict_fn()
101
102        if self.background_data is not None:
103            n = min(self.n_background_samples, len(self.background_data))
104            bg = shap.sample(self.background_data, n)
105        else:
106            n = min(self.n_background_samples, len(self._X_fit))
107            bg = shap.sample(self._X_fit, n)
108
109        self.explainer_ = shap.PermutationExplainer(predict_fn, bg)
110        explanation = self.explainer_(self._X_fit)
111
112        raw_values = explanation.values
113
114        # Normalise binary classification: take positive-class SHAP values
115        if isinstance(raw_values, list):
116            raw_values = raw_values[1] if len(raw_values) == 2 else raw_values[0]
117        elif isinstance(raw_values, np.ndarray) and raw_values.ndim == 3:
118            raw_values = raw_values[:, :, 1]
119
120        self.shap_values_ = np.array(raw_values)
121
122        # Extract base value from first sample (same for all samples)
123        base = np.array(explanation.base_values)
124        if base.ndim == 2:
125            # (n_samples, n_classes) — binary classification: take positive class
126            self.base_value_ = float(base[0, 1]) if base.shape[1] >= 2 else float(base[0, 0])
127        elif base.ndim == 1:
128            self.base_value_ = float(base[0])
129        else:
130            self.base_value_ = float(base)
131
132        self._is_fitted = True
133        return self

def get_feature_importance(self) -> pandas.core.frame.DataFrame: 
135    def get_feature_importance(self) -> pd.DataFrame:
136        """Return global feature importance based on mean absolute SHAP values.
137
138        Returns
139        -------
140        pd.DataFrame
141            Columns: rank, feature, importance_pct, direction, mean_abs_shap.
142
143            - rank: importance ranking (1 = most important)
144            - feature: feature name
145            - importance_pct: normalized importance (0-1 scale, sums to 1)
146            - direction: ``"positive"`` if feature generally increases prediction,
147              ``"negative"`` if it decreases prediction (based on mean SHAP value)
148            - mean_abs_shap: mean absolute SHAP value across all samples
149
150            Sorted by importance descending.
151
152        Raises
153        ------
154        RuntimeError
155            If called before fit.
156        """
157        self._check_fitted()
158        mean_abs = np.abs(self.shap_values_).mean(axis=0)
159        mean_shap = self.shap_values_.mean(axis=0)
160
161        total_importance = mean_abs.sum()
162        importance_pct = mean_abs / total_importance if total_importance > 0 else mean_abs
163
164        df = pd.DataFrame(
165            {
166                "feature": self.feature_names_,
167                "mean_abs_shap": mean_abs,
168                "importance_pct": importance_pct,
169                "direction": ["positive" if v >= 0 else "negative" for v in mean_shap],
170            }
171        )
172        df = df.sort_values("mean_abs_shap", ascending=False).reset_index(drop=True)
173        df["rank"] = np.arange(1, len(df) + 1)
174        return df[["rank", "feature", "importance_pct", "direction", "mean_abs_shap"]]

190            - feature: feature name

211        shap_vals = self.shap_values_[row_index]
212        abs_shap = np.abs(shap_vals)

216
217        df = pd.DataFrame(
218            {
219                "feature": self.feature_names_,
220                "feature_value": feature_values,
221                "shap_value": shap_vals,
222                "importance_pct": importance_pct,
223                "direction": ["positive" if v >= 0 else "negative" for v in shap_vals],
224            }
225        )
226        df = df.sort_values("importance_pct", ascending=False).reset_index(drop=True)
227        df["rank"] = np.arange(1, len(df) + 1)
