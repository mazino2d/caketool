  9class AdversarialModel:
 10    """
 11    A class designed to detect drift between two datasets by training a binary
 12    classifier to distinguish between them. Drift is detected when the model can
 13    successfully differentiate between the two datasets, indicating that they
 14    come from different distributions.
 15
 16    Attributes:
 17    -----------
 18    model : XGBClassifier
 19        XGBoost classifier used to detect drift.
 20    auc_score : float
 21        The ROC AUC score representing the ability of the model to distinguish
 22        between the two datasets (high score indicates drift).
 23
 24    Methods:
 25    --------
 26    fit(df1, df2, groups_col=["label"], features=None):
 27        Fits the adversarial classifier to the concatenated data to detect drift.
 28    show(n_features=5):
 29        Displays the ROC AUC score and the top N most important features contributing to drift.
 30    """
 31
 32    def __init__(self) -> None:
 33        """Initializes the AdversarialModel with XGBClassifier and TargetEncoder."""
 34        self.model = xgb.XGBClassifier(verbosity=0, eval_metric="auc")
 35        self.encoder = FeatureEncoder("category_encoders.TargetEncoder")
 36        self.auc_score = -1
 37
 38    def fit(
 39        self,
 40        df1: pd.DataFrame,
 41        df2: pd.DataFrame,
 42        groups_col=None,
 43        features=None,
 44    ):
 45        """
 46        Fits the adversarial classifier to detect drift between two datasets by
 47        differentiating between them. If the model can classify data points
 48        from df1 and df2 effectively, this indicates that drift has occurred.
 49
 50        Parameters:
 51        -----------
 52        df1 : pd.DataFrame
 53            The first dataset representing the reference distribution (e.g., training data).
 54        df2 : pd.DataFrame
 55            The second dataset to compare against the reference (e.g., new or testing data).
 56        groups_col : list, optional
 57            Column(s) used for stratification during the train-test split (default is ["label"]).
 58        features : list, optional
 59            List of feature names to be used for training (default is the intersection of the columns in both datasets).
 60
 61        Returns:
 62        --------
 63        None
 64        """
 65        if groups_col is None:
 66            groups_col = ["label"]
 67        # Concatenating the two datasets and assigning labels (0 for df1, 1 for df2)
 68        data_adversarial = pd.concat(
 69            [
 70                df1.assign(label=0),
 71                df2.assign(label=1),
 72            ],
 73            ignore_index=True,
 74        )
 75
 76        # If features are not provided, use the common columns between the two datasets
 77        if features is None:
 78            features = list(set(df1.columns).intersection(df2.columns))
 79
 80        # Splitting the data into training and validation sets using stratified sampling
 81        X_train, X_val, y_train, y_val = train_test_split(
 82            data_adversarial[features],
 83            data_adversarial["label"],
 84            test_size=0.2,
 85            stratify=data_adversarial[groups_col].apply(lambda x: "_".join(map(str, x)), axis=1),
 86        )
 87
 88        # Encode categorical features
 89        X_train = self.encoder.fit_transform(X_train, y_train)
 90        X_val = self.encoder.transform(X_val)
 91        self.feature_names_ = X_train.columns.tolist()
 92
 93        # Fitting the model
 94        self.model.fit(X_train, y_train)
 95
 96        # Calculating the ROC AUC score to measure the ability to detect drift
 97        self.auc_score = roc_auc_score(y_val, self.model.predict_proba(X_val)[:, 1])
 98
 99    def show(self, n_features=5):
100        """
101        Displays the ROC AUC score of the model, which represents the ability
102        to detect drift between the two datasets. It also shows the top N
103        important features contributing to drift detection.
104
105        Parameters:
106        -----------
107        n_features : int, optional
108            The number of top important features to display (default is 5).
109
110        Returns:
111        --------
112        None
113        """
114        print(f"ROC AUC: {self.auc_score:02f}")
115        print(f"Top {n_features} important feature(s) contributing to drift:")
116        importance_df = pd.DataFrame(
117            {
118                "feature": self.feature_names_,
119                "importance": self.model.feature_importances_,
120            }
121        ).sort_values("importance", ascending=False)
122        print(importance_df.head(n_features).to_string(index=False))
