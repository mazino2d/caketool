  8class ExperimentTracker(ABC):
  9    """
 10    Abstract base class for experiment tracking.
 11
 12    Provides a unified interface for logging parameters, metrics, and artifacts
 13    to different experiment tracking backends (Vertex AI, MLflow, wandb).
 14
 15    Use this class as a context manager (``with`` statement) to automatically
 16    open and close a run. Set ``mode="deploy"`` to disable all logging in
 17    production without changing any other code.
 18
 19    API keys and credentials should be set via environment variables, either
 20    directly in the shell or through a ``.env`` file passed via ``dotenv_path``.
 21
 22    Parameters
 23    ----------
 24    experiment_name : str
 25        Name of the experiment. Groups multiple runs together.
 26    run_name : str
 27        Name of this specific run within the experiment.
 28    mode : {"develop", "deploy"}, optional
 29        ``"develop"`` enables logging (default). ``"deploy"`` silently skips
 30        all logging — useful for production inference.
 31    dotenv_path : str or None, optional
 32        Path to a ``.env`` file to load before initializing the tracker.
 33        Useful for storing API keys and credentials outside the codebase.
 34    """
 35
 36    def __init__(
 37        self,
 38        experiment_name: str,
 39        run_name: str,
 40        mode: Literal["develop", "deploy"] = "develop",
 41        dotenv_path: str | None = None,
 42    ) -> None:
 43        if dotenv_path is not None:
 44            from dotenv import load_dotenv
 45
 46            load_dotenv(dotenv_path)
 47        self.experiment_name = experiment_name
 48        self.run_name = run_name
 49        self.mode = mode

 58
 59        Parameters
 60        ----------
 61        params : dict[str, float | int | str]
 62            Hyperparameter names and their values.
 63        """
 64        pass

 68        """
 69        Log evaluation metrics for this run.
 70
 71        Parameters
 72        ----------
 73        metrics : dict[str, float | int | str]

139    """
140    Experiment tracker backed by Google Cloud Vertex AI.
141
142    Authenticates via Application Default Credentials (ADC). Either run
143    ``gcloud auth application-default login``, or set
144    ``GOOGLE_APPLICATION_CREDENTIALS`` to the path of a service account key
145    file in your ``.env``.
146
147    Example ``.env`` file::
148
149        GOOGLE_APPLICATION_CREDENTIALS=/path/to/service-account-key.json
150        GOOGLE_CLOUD_PROJECT=my-gcp-project
151        GOOGLE_CLOUD_LOCATION=us-central1
152        VERTEX_STAGING_BUCKET=gs://my-bucket
153
154    Parameters
155    ----------
156    experiment_name : str
157        Name of the Vertex AI experiment.
158    run_name : str
159        Name of this run within the experiment.
160    project : str or None, optional
161        GCP project ID. Falls back to ``GOOGLE_CLOUD_PROJECT`` env var.
162    location : str or None, optional
163        GCP region (e.g. ``"us-central1"``). Falls back to
164        ``GOOGLE_CLOUD_LOCATION`` env var.
165    bucket_name : str or None, optional
166        GCS bucket URI for artifact storage (e.g. ``"gs://my-bucket"``).
167        Falls back to ``VERTEX_STAGING_BUCKET`` env var.
168    mode : {"develop", "deploy"}, optional
169        ``"develop"`` enables logging (default). ``"deploy"`` disables it.
170    experiment_description : str or None, optional
171        Short description of what this experiment is testing.
172    experiment_tensorboard : bool, optional
173        Whether to enable TensorBoard integration (default: ``False``).
174    dotenv_path : str or None, optional
175        Path to a ``.env`` file. Cannot be combined with ``project``,
176        ``location``, or ``bucket_name`` if their corresponding env vars are
177        also present in the file (raises ``ValueError``).
178    """
179
180    @require_dependencies("google.cloud.aiplatform", "google.cloud.storage")
181    def __init__(
182        self,
183        experiment_name: str,
184        run_name: str,
185        project: str | None = None,
186        location: str | None = None,
187        bucket_name: str | None = None,
188        mode: Literal["develop", "deploy"] = "develop",
189        experiment_description: str | None = None,
190        experiment_tensorboard: bool = False,
191        dotenv_path: str | None = None,
192    ) -> None:
193        import os
194
195        from google.cloud import aiplatform, storage
196
197        _env_map = {
198            "project": "GOOGLE_CLOUD_PROJECT",
199            "location": "GOOGLE_CLOUD_LOCATION",
200            "bucket_name": "VERTEX_STAGING_BUCKET",
201        }
202        if dotenv_path is not None:
203            from dotenv import dotenv_values
204
205            env_vals = dotenv_values(dotenv_path)
206            conflicts = [p for p, e in _env_map.items() if locals()[p] is not None and e in env_vals]
207            if conflicts:
208                raise ValueError(f"Cannot pass both dotenv_path and direct params: {conflicts}")
209
210        super().__init__(experiment_name, run_name, mode, dotenv_path=dotenv_path)
211
212        self.project = project or os.environ.get("GOOGLE_CLOUD_PROJECT")
213        self.location = location or os.environ.get("GOOGLE_CLOUD_LOCATION")
214        self.bucket_name = bucket_name or os.environ.get("VERTEX_STAGING_BUCKET")
215        if not self.project:
216            raise ValueError("project must be provided or set as GOOGLE_CLOUD_PROJECT in .env")
217        if not self.location:
218            raise ValueError("location must be provided or set as GOOGLE_CLOUD_LOCATION in .env")
219        if not self.bucket_name:
220            raise ValueError("bucket_name must be provided or set as VERTEX_STAGING_BUCKET in .env")
221

223        self.experiment_tensorboard = experiment_tensorboard
224        self.gs_client = storage.Client(project=self.project)

227        self._storage = storage

229        aiplatform.init(
230            project=self.project,
231            location=self.location,
232            experiment=self.experiment_name,
233            experiment_description=self.experiment_description,
234            experiment_tensorboard=self.experiment_tensorboard,
235            staging_bucket=self.bucket_name,
236        )
237        if self.mode == "develop":
238            self.experiment_run = aiplatform.start_run(self.run_name)
239            self.execution = aiplatform.start_execution(
240                display_name="Experiment Tracking", schema_title="system.ContainerExecution"
241            )

244        """
245        Log hyperparameters to the Vertex AI experiment run.

247        Parameters

255    def log_metrics(self, metrics: dict[str, float | int | str], step: int | None = None) -> None:
256        """
257        Log metrics to the Vertex AI experiment run.
258
259        When ``step`` is provided, uses ``log_time_series_metrics`` to record
260        metrics at a specific training step (e.g., for plotting learning curves).
261
262        Parameters
263        ----------
264        metrics : dict[str, float | int | str]
265            Metric names and their values.
266        step : int or None, optional
267            Training step or epoch number.
268        """
269        if self.mode == "develop":
270            if step is not None:
271                self.experiment_run.log_time_series_metrics(metrics, step=step)
272            else:
273                self.experiment_run.log_metrics(metrics)
274
275    def log_file(self, filename: str, artifact_id: str) -> None:
276        """
277        Upload a file to GCS and register it as a Vertex AI artifact.
278
279        Parameters
280        ----------
281        filename : str
282            Local path to the file to upload.
283        artifact_id : str
284            Unique identifier for this artifact within the run.
285        """
286        if self.mode == "develop":
287            blob = self._add_artifact(artifact_id)
288            blob.upload_from_filename(filename)
289
290    def log_pickle(self, var: object, artifact_id: str) -> None:
291        """
292        Pickle an object and upload it directly to GCS (no temp file needed).
293
294        Parameters
295        ----------
296        var : object
297            Any picklable Python object.
298        artifact_id : str
299            Unique identifier for this artifact within the run.
300        """
301        if self.mode == "develop":
302            pickle_out = pickle.dumps(var)
303            blob = self._add_artifact(artifact_id)
304            blob.upload_from_string(pickle_out)
305
306    def load_pickle(self, artifact_id: str) -> object:
307        """
308        Download an artifact from GCS and unpickle it.
309
310        Parameters
311        ----------
312        artifact_id : str
313            Identifier of the artifact to load.
314
315        Returns
316        -------
317        object
318            The unpickled Python object.
319        """
320        blob = self._get_blob(artifact_id)
321        pickle_in = blob.download_as_string()
322        return pickle.loads(pickle_in)
323
324    def _get_blob(self, artifact_id: str):
325        """Return the GCS blob for the given artifact (does not check existence)."""
326        blob_name = f"{self.experiment_name}-{self.run_name}-{artifact_id}"
327        blob = self.gc_bucket.blob(blob_name)
328        return blob
329
330    def _add_artifact(self, artifact_id: str):
331        """
332        Register a new artifact in Vertex AI and return its GCS blob.
333
334        Raises ``ValueError`` if the artifact already exists to prevent
335        accidental overwrites.
336        """
337        blob = self._get_blob(artifact_id)
338        uri = blob.path_helper(self.bucket_name, blob.name)
339        if blob.exists():
340            raise ValueError(f"{uri} existed! (Cannot overwrite)")
341        artifact = self._aiplatform.Artifact.create(uri=uri, schema_title="system.Artifact")
342        self.experiment_run._metadata_node.add_artifacts_and_executions(
343            artifact_resource_names=[artifact.resource_name]
344        )
345        return blob
346
347    def __enter__(self):
348        """Open the Vertex AI run and execution context."""
349        if self.mode == "develop":
350            self.execution.__enter__()
351            self.experiment_run.__enter__()
352        return self
353
354    def __exit__(self, exc_type, exc_value, exc_traceback):
355        """Close the Vertex AI run and execution context."""
356        if self.mode == "develop":
357            self.execution.__exit__(exc_type, exc_value, exc_traceback)
358            self.experiment_run.__exit__(exc_type, exc_value, exc_traceback)

361class MLflowTracker(ExperimentTracker):
362    """
363    Experiment tracker backed by MLflow.
364
365    Connects to an MLflow tracking server. For authenticated servers, embed
366    credentials in the URI (e.g. ``http://user:password@host:5000``) or set
367    ``MLFLOW_TRACKING_USERNAME`` / ``MLFLOW_TRACKING_PASSWORD`` as env vars.
368
369    The experiment is created automatically if it does not already exist.
370
371    Example ``.env`` file::
372
373        MLFLOW_TRACKING_URI=http://localhost:5000
374        MLFLOW_TRACKING_USERNAME=my-user
375        MLFLOW_TRACKING_PASSWORD=my-password
376        MLFLOW_ARTIFACT_LOCATION=s3://my-bucket/artifacts
377
378    Parameters
379    ----------
380    experiment_name : str
381        Name of the MLflow experiment. Created if it does not exist.
382    run_name : str
383        Name of this run within the experiment.
384    tracking_uri : str or None, optional
385        URI of the MLflow tracking server. Falls back to
386        ``MLFLOW_TRACKING_URI`` env var. Uses the local filesystem if unset.
387    artifact_location : str or None, optional
388        Storage location for artifacts (only applied when creating a new
389        experiment). Falls back to ``MLFLOW_ARTIFACT_LOCATION`` env var.
390    mode : {"develop", "deploy"}, optional
391        ``"develop"`` enables logging (default). ``"deploy"`` disables it.
392    tags : dict[str, str] or None, optional
393        Key-value tags to attach to the run (e.g. ``{"team": "ml"}``).
394    dotenv_path : str or None, optional
395        Path to a ``.env`` file. Cannot be combined with ``tracking_uri`` or
396        ``artifact_location`` if their env vars are also in the file.
397    """
398
399    @require_dependencies("mlflow")
400    def __init__(
401        self,
402        experiment_name: str,
403        run_name: str,
404        tracking_uri: str | None = None,
405        artifact_location: str | None = None,
406        mode: Literal["develop", "deploy"] = "develop",
407        tags: dict[str, str] = None,
408        dotenv_path: str | None = None,
409    ) -> None:
410        import os
411
412        import mlflow
413
414        _env_map = {"tracking_uri": "MLFLOW_TRACKING_URI", "artifact_location": "MLFLOW_ARTIFACT_LOCATION"}
415        if dotenv_path is not None:
416            from dotenv import dotenv_values
417
418            env_vals = dotenv_values(dotenv_path)
419            conflicts = [p for p, e in _env_map.items() if locals()[p] is not None and e in env_vals]
420            if conflicts:
421                raise ValueError(f"Cannot pass both dotenv_path and direct params: {conflicts}")
422
423        super().__init__(experiment_name, run_name, mode, dotenv_path=dotenv_path)
424
425        self.tracking_uri = tracking_uri or os.environ.get("MLFLOW_TRACKING_URI")
426        self.artifact_location = artifact_location or os.environ.get("MLFLOW_ARTIFACT_LOCATION")
427        self.tags = tags or {}
428        self._mlflow = mlflow
429        self._run = None
430
431        if self.tracking_uri:
432            mlflow.set_tracking_uri(self.tracking_uri)
433
434        # Create or get experiment
435        experiment = mlflow.get_experiment_by_name(experiment_name)
436        if experiment is None:
437            self.experiment_id = mlflow.create_experiment(experiment_name, artifact_location=self.artifact_location)
438        else:
439            self.experiment_id = experiment.experiment_id
440
441    def log_params(self, params: dict[str, float | int | str]) -> None:
442        """
443        Log hyperparameters to the active MLflow run.
444
445        Parameters
446        ----------
447        params : dict[str, float | int | str]
448            Hyperparameter names and their values.
449        """
450        if self.mode == "develop" and self._run:
451            self._mlflow.log_params(params)
452
453    def log_metrics(self, metrics: dict[str, float | int | str], step: int | None = None) -> None:
454        """
455        Log metrics to the active MLflow run.
456
457        Parameters
458        ----------
459        metrics : dict[str, float | int | str]
460            Metric names and their values.
461        step : int or None, optional
462            Training step or epoch number. Used for plotting metrics over time.
463        """
464        if self.mode == "develop" and self._run:
465            self._mlflow.log_metrics(metrics, step=step)
466
467    def log_file(self, filename: str, artifact_id: str) -> None:
468        """
469        Upload a file to the MLflow artifact store.
470
471        Parameters
472        ----------
473        filename : str
474            Local path to the file to upload.
475        artifact_id : str
476            Subdirectory path within the run's artifact store.
477        """
478        if self.mode == "develop" and self._run:
479            self._mlflow.log_artifact(filename, artifact_path=artifact_id)
480
481    def log_pickle(self, var: object, artifact_id: str) -> None:
482        """
483        Pickle an object to a temp file and upload it to the MLflow artifact store.
484
485        Parameters
486        ----------
487        var : object
488            Any picklable Python object.
489        artifact_id : str
490            Name used for the ``.pkl`` file in the artifact store.
491        """
492        if self.mode == "develop" and self._run:
493            import os
494            import tempfile
495
496            with tempfile.TemporaryDirectory() as tmpdir:
497                filepath = os.path.join(tmpdir, f"{artifact_id}.pkl")
498                with open(filepath, "wb") as f:
499                    pickle.dump(var, f)
500                self._mlflow.log_artifact(filepath)
501
502    def load_pickle(self, artifact_id: str) -> object:
503        """
504        Download and unpickle an artifact from the MLflow artifact store.
505
506        Parameters
507        ----------
508        artifact_id : str
509            Name of the artifact to load (without ``.pkl`` extension).
510
511        Returns
512        -------
513        object
514            The unpickled Python object.
515
516        Raises
517        ------
518        RuntimeError
519            If called outside a ``with`` block (no active run).
520        """
521        if self._run is None:
522            raise RuntimeError("No active run. Use within context manager or start a run first.")
523
524        artifact_uri = f"{self._run.info.artifact_uri}/{artifact_id}.pkl"
525        local_path = self._mlflow.artifacts.download_artifacts(artifact_uri)
526        with open(local_path, "rb") as f:
527            return pickle.load(f)
528
529    def __enter__(self):
530        """Start a new MLflow run."""
531        if self.mode == "develop":
532            self._run = self._mlflow.start_run(
533                experiment_id=self.experiment_id,
534                run_name=self.run_name,
535                tags=self.tags,
536            )
537        return self
538
539    def __exit__(self, exc_type, exc_value, exc_traceback):
540        """End the active MLflow run."""
541        if self.mode == "develop" and self._run:
542            self._mlflow.end_run()
543            self._run = None
