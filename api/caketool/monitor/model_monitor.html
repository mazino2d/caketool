 11class ModelMonitor:
 12    """BigQuery-backed monitor for tracking feature and score distributions.
 13
 14    Stores reference distributions (bin thresholds and per-bin counts) in
 15    BigQuery tables and provides utilities to normalise data, create binning
 16    schemas, and calculate current distributions for comparison.
 17
 18    The typical workflow is:
 19
 20    1. ``normalize_data`` – Standardise dtypes and fill/cap missing/infinity
 21       values so that all downstream steps receive clean data.
 22    2. ``create_bin_data`` – Compute percentile-based bin edges for numeric
 23       features and top-K category lists for categorical features.
 24    3. ``store_bin_data`` – Persist the bin schema to BigQuery.
 25    4. ``calc_feature_distribution`` – Compute per-bin counts for a dataset
 26       using the stored bin schema.
 27    5. ``store_feature_distribution`` – Persist the distribution to BigQuery.
 28    6. Later, compare stored distributions using PSI or similar metrics.
 29
 30    Parameters
 31    ----------
 32    project : str
 33        Google Cloud project ID used for BigQuery operations.
 34    location : str
 35        BigQuery dataset location (e.g. ``"US"`` or ``"asia-southeast1"``).
 36    dataset : str, optional
 37        BigQuery dataset name where monitoring tables are created.
 38        Defaults to ``"model_motinor"`` (note: intentional legacy spelling).
 39
 40    Attributes
 41    ----------
 42    MISSING : str
 43        Sentinel string used to represent missing / null categories
 44        (``"cake.miss"``).
 45    OTHER : str
 46        Sentinel string used to represent out-of-vocabulary categories
 47        (``"cake.other"``).
 48    bq_client : bigquery.Client
 49        Authenticated BigQuery client.
 50    ID_COLS : list[str]
 51        Fixed identifier columns added to every monitoring table:
 52        ``["score_type", "dataset_type", "version_type", "version"]``.
 53    """

 65    def normalize_data(
 66        self, df: pd.DataFrame, inplace: bool = False, cate_missing_values: set[str] | None = None
 67    ) -> pd.DataFrame:
 68        """Standardise data types and encode missing / infinity values.
 69
 70        Applies the following transformations in order:
 71
 72        1. Fills ``NaN`` with ``-100`` (numeric sentinel).
 73        2. Coerces columns to numeric where possible; remaining columns are
 74           treated as categorical (converted to lowercase strings).
 75        3. Replaces ``+inf`` with ``-100`` in numeric columns.
 76        4. Casts numeric columns to ``int`` or ``float`` based on whether
 77           fractional values are present.
 78        5. Replaces any negative numeric value with ``-100``.
 79        6. Replaces categorical values in *cate_missing_values* with
 80           ``self.MISSING``; strips Vietnamese diacritics and lowercases
 81           all other categorical values.
 82
 83        A ``__is_norm`` attribute is set on the returned DataFrame so that
 84        downstream methods can confirm normalisation has been applied.
 85
 86        Parameters
 87        ----------
 88        df : pd.DataFrame
 89            Raw input features.
 90        inplace : bool, optional
 91            If ``True``, modifies *df* in place and returns it.  If
 92            ``False`` (default), operates on a copy.
 93        cate_missing_values : set[str] or None, optional
 94            Set of string values that should be treated as missing for
 95            categorical columns.  Defaults to
 96            ``{"-1", "-100", "unknown", ""}``.
 97
 98        Returns
 99        -------
100        pd.DataFrame
101            Normalised DataFrame with ``__is_norm = True`` set as an
102            attribute.
103        """
104        if cate_missing_values is None:
105            cate_missing_values = {"-1", "-100", "unknown", ""}
106        if not inplace:
107            df = df.copy()
108        # Fill missing value
109        df = df.fillna(-100)
110        # Norm categorical feature type
111        categorical_features: list[str] = []
112        for col in df.columns:
113            try:
114                df[col] = pd.to_numeric(df[col])
115            except (ValueError, KeyError):
116                categorical_features.append(col)
117        df[categorical_features] = df[categorical_features].apply(lambda x: x.astype(str).str.lower())
118
119        # Find numerical features
120        numerical_features: list[str] = df.select_dtypes([int, float]).columns
121        # Handle infinity value
122        for col in list(set(df[numerical_features].columns.to_series()[np.isinf(df[numerical_features]).any()])):
123            df[col] = df[col].apply(lambda x: -100 if x == np.inf else x)
124        # Norm float feature
125        float_features: list[str] = []
126        for col in numerical_features:
127            if df[col].fillna(0).nunique() > round(df[col].fillna(0)).nunique():
128                float_features.append(col)
129        df[float_features] = df[float_features].astype(float)
130        # Norm int feature
131        int_features = list(set(df.columns) - set(categorical_features) - set(float_features))
132        df[int_features] = df[int_features].astype(int)
133        # Handle numeric columns
134        for col in numerical_features:
135            df[col] = df[col].apply(lambda x: -100 if x < 0 else x)
136        # Handle categorical columns
137        for col in categorical_features:
138            df[col] = df[col].apply(
139                lambda x: (
140                    self.MISSING if x in cate_missing_values else str_utils.remove_vn_diacritics(x).lower().strip()
141                )
142            )
143        df.__is_norm = True
144        return df

146    def create_bin_data(self, df: pd.DataFrame, n_bins=10) -> dict[str, list[float | str]]:
147        """Compute bin thresholds for all features in *df*.
148
149        For **numeric** columns, generates ``n_bins`` percentile-based
150        edges from the positive (non-missing) values.  A leading ``0.0``
151        edge is always prepended so that the first bin captures the
152        zero/missing mass, and the final edge is nudged up by ``1e-10``
153        to make the last interval closed on the right.
154
155        For **categorical** columns (object dtype), collects the top
156        ``n_bins`` most frequent non-missing categories plus sentinel
157        values ``self.MISSING`` and ``self.OTHER``.
158
159        Parameters
160        ----------
161        df : pd.DataFrame
162            Normalised DataFrame (must have been processed by
163            ``normalize_data`` first).
164        n_bins : int, optional
165            Number of percentile bins for numeric features and maximum
166            number of categories to retain for categorical features.
167            Defaults to ``10``.
168
169        Returns
170        -------
171        pd.DataFrame
172            DataFrame with columns ``["feature_name", "type", "bins"]``:
173
174            - ``feature_name`` – Column name.
175            - ``type`` – Data type string (e.g. ``"float64"``, ``"int64"``,
176              ``"string"``).
177            - ``bins`` – List of numeric bin edges (float) for numeric
178              features, or list of category strings for categorical features.
179
180            Rows are sorted alphabetically by ``feature_name``.
181
182        Raises
183        ------
184        Exception
185            If *df* has not been normalised with ``normalize_data`` first.
186        """
187        if not self._check_norm(df):
188            raise Exception("DataFrame has not been normalized yet. Please use self.normalize_data(df)")
189        numerical_features = set(df.select_dtypes([int, float]).columns)
190        categorical_features = list(df.select_dtypes([object]).columns)
191        bin_thresholds = []
192        # Bin num features
193        for f in numerical_features:
194            series = df[f]
195            series = series[(series > 0) & (~series.isna())]
196            if len(series) == 0:
197                bins = np.array([])
198            else:
199                percentage = np.linspace(0, 100, n_bins + 1)
200                bins = np.percentile(series, percentage)
201                bins = [num_utils.round(e, series.dtype) for e in bins]
202                bins = np.unique([0.0, *bins])
203                if len(bins) >= 2:
204                    bins[-1] = bins[-1] + 1e-10
205            bin_thresholds.append([f, str(series.dtype).lower(), bins.tolist()])
206        # Bin cate features
207        for f in categorical_features:
208            series = df[f]
209            bins = series[series != self.MISSING].value_counts()[:n_bins].index.to_list()
210            bins = sorted(set([self.MISSING, self.OTHER, *bins]))
211            bin_thresholds.append([f, "string", bins])
212
213        df_bins = pd.DataFrame(bin_thresholds, columns=["feature_name", "type", "bins"])
214        return df_bins.sort_values("feature_name").reset_index(drop=True)

264    def load_bin_data(
265        self,
266        score_type: str,
267        dataset_type: str,
268        version_type: str,
269        version: str,
270        bq_table_name: str,
271    ):
272        """Load a bin-threshold table from BigQuery.
273
274        Fetches the bin schema previously stored by ``store_bin_data`` and
275        restores numeric bin edges to ``float`` (categorical bins remain
276        as string lists).
277
278        Parameters
279        ----------
280        score_type : str
281            Model or score identifier.
282        dataset_type : str
283            Dataset split label.
284        version_type : str
285            Version dimension.
286        version : str
287            Specific version value.
288        bq_table_name : str
289            Fully qualified BigQuery table name.
290
291        Returns
292        -------
293        pd.DataFrame
294            Bin-threshold DataFrame with columns
295            ``["feature_name", "type", "bins"]``, sorted alphabetically
296            by ``feature_name``.  Numeric bins are lists of ``float``;
297            categorical bins are lists of ``str``.
298        """
299        df_bins: pd.DataFrame = self.bq_client.query(f"""
300            SELECT feature_name, type, bins FROM {bq_table_name}
301            WHERE score_type = '{score_type}'
302            AND dataset_type = '{dataset_type}'
303            AND version_type = '{version_type}'
304            AND version = '{version}'
305        """).to_dataframe()
306
307        def cvt_bins(r):
308            if r["type"] == "string":
309                return r["bins"]
310            else:
311                return [float(e) for e in r["bins"]]
312
313        df_bins["bins"] = df_bins.apply(cvt_bins, axis=1)
314        return df_bins.sort_values("feature_name").reset_index(drop=True)

316    def calc_feature_distribution(self, df: pd.DataFrame, df_bins: dict[str, np.ndarray]) -> pd.DataFrame:
317        """Compute per-bin counts for each feature using the provided bin schema.
318
319        For **numeric** features, uses ``np.histogram`` with ``[-inf, *bins, +inf]``
320        boundaries so that out-of-range values are captured in the first
321        (``"missing"``) and last (``"other"``) segments.
322
323        For **categorical** features, maps any value not present in the
324        stored category list to ``self.OTHER`` and counts occurrences per
325        category.
326
327        Segment labels follow the pattern ``"A. [lo, hi)"`` (numeric) or
328        ``"A. category_name"`` (categorical), where the letter prefix
329        (A–Z) allows consistent ordering when displayed in BI tools.
330
331        Parameters
332        ----------
333        df : pd.DataFrame
334            Normalised DataFrame (must have been processed by
335            ``normalize_data`` first).
336        df_bins : pd.DataFrame
337            Bin-threshold table as returned by ``load_bin_data`` or
338            ``create_bin_data``.
339
340        Returns
341        -------
342        pd.DataFrame
343            Exploded distribution table with columns:
344
345            - ``feature_name`` – Feature column name.
346            - ``segment`` – Labelled bin string (e.g. ``"A. [0, 100)"``).
347            - ``count`` – Number of observations in this segment.
348            - ``total`` – Total observations for this feature.
349            - ``percent`` – Fraction of observations in this segment.
350
351        Raises
352        ------
353        Exception
354            If *df* has not been normalised with ``normalize_data`` first.
355        """
356        if not self._check_norm(df):
357            raise Exception("DataFrame has not been normalized yet. Please use self.normalize_data(df)")
358        bin_thresholds: dict[str, np.ndarray] = dict(zip(df_bins.feature_name, df_bins.bins, strict=True))
359        hists = []
360        categorical_features = [k for k, v in bin_thresholds.items() if len(v) > 0 and isinstance(v[0], str)]
361        numerical_features = [k for k, v in bin_thresholds.items() if len(v) > 0 and not isinstance(v[0], str)]
362
363        for f in numerical_features:
364            series = df[f]
365            hist, _ = np.histogram(series, [-np.inf, *bin_thresholds[f], np.inf])
366            segments = ["missing", *self._cvt_bins2labels(bin_thresholds[f]), "other"]
367            segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, segments, strict=False)]
368            hists.append([f, segments, hist, hist.sum(), hist / hist.sum()])
369        for f in categorical_features:
370            series = df[f]
371            bins = bin_thresholds[f]
372            series = series.apply(lambda x, b=bins: x if x in b else self.OTHER)
373            vc = series.value_counts()
374            for bin_name in bins:
375                if bin_name not in vc.index:
376                    vc.loc[bin_name] = 0
377            vc = vc.reindex(bins)
378            segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, bins, strict=False)]
379            hists.append([f, segments, vc, vc.sum(), vc / vc.sum()])
380
381        return pd.DataFrame(hists, columns=["feature_name", "segment", "count", "total", "percent"]).explode(
382            ["segment", "count", "percent"]
383        )

410    def calc_score_distribution(self, score: np.ndarray, bins: int | list[float] = 10):
411        """Compute the distribution of a score array over percentile-based bins.
412
413        Parameters
414        ----------
415        score : np.ndarray
416            1-D array of predicted scores (e.g. model probabilities).
417        bins : int or list[float], optional
418            If an integer, creates that many equal-frequency bins via
419            ``create_percentile_bins``.  If a list of floats, uses them
420            directly as bin edges.  Defaults to ``10``.
421
422        Returns
423        -------
424        pd.DataFrame
425            Distribution table with columns:
426
427            - ``segment`` – Labelled interval string (e.g. ``"A. [0.0, 0.1)"``)
428            - ``count`` – Number of scores in this bin.
429            - ``total`` – Total number of scores.
430            - ``percent`` – Fraction of scores in this bin.
431        """
432        if isinstance(bins, int):
433            bins = arr_utils.create_percentile_bins(score, bins)
434        total = len(score)
435        histogram = np.histogram(score, bins)[0]
436        percent = histogram / total
437        segments = self._cvt_bins2labels(bins)
438        segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, segments, strict=False)]
439        return pd.DataFrame(
440            {
441                "segment": segments,
442                "count": histogram,
443                "total": [total] * len(histogram),
444                "percent": percent,
445            }
446        )

499        self,
500        score_type: str,
501        dataset_type: str,
502        version_type: str,
503        version: str,
504        df: pd.DataFrame,

507    ):
508        job_config = bigquery.LoadJobConfig(

518        self._clear_data(bq_table_name, score_type, dataset_type, version_type, version)
519        return self.bq_client.load_table_from_dataframe(df, bq_table_name, job_config=job_config).result()

521    def _cvt_bins2labels(self, bins: list[object]) -> list[str]:
522        if len(bins) <= 1:
523            return []
524        bins = [round(float(e), 2) for e in bins]
525        bins = [int(e) if e.is_integer() else e for e in bins]
526        segments = ["[" + ", ".join(map(str, e)) + ")" for e in zip(bins[:-1], bins[1:], strict=True)]
527        segments[-1] = segments[-1][:-1] + "]"
528        return segments

 95            Set of string values that should be treated as missing for
 96            categorical columns.  Defaults to
 97            ``{"-1", "-100", "unknown", ""}``.
 98
 99        Returns
100        -------
101        pd.DataFrame
102            Normalised DataFrame with ``__is_norm = True`` set as an
103            attribute.
104        """
105        if cate_missing_values is None:
106            cate_missing_values = {"-1", "-100", "unknown", ""}
107        if not inplace:
108            df = df.copy()
109        # Fill missing value
110        df = df.fillna(-100)
111        # Norm categorical feature type
112        categorical_features: list[str] = []
113        for col in df.columns:
114            try:
115                df[col] = pd.to_numeric(df[col])
116            except (ValueError, KeyError):
117                categorical_features.append(col)
118        df[categorical_features] = df[categorical_features].apply(lambda x: x.astype(str).str.lower())
119
120        # Find numerical features
121        numerical_features: list[str] = df.select_dtypes([int, float]).columns
122        # Handle infinity value
123        for col in list(set(df[numerical_features].columns.to_series()[np.isinf(df[numerical_features]).any()])):
124            df[col] = df[col].apply(lambda x: -100 if x == np.inf else x)
125        # Norm float feature
126        float_features: list[str] = []
127        for col in numerical_features:
128            if df[col].fillna(0).nunique() > round(df[col].fillna(0)).nunique():
129                float_features.append(col)
130        df[float_features] = df[float_features].astype(float)
131        # Norm int feature
132        int_features = list(set(df.columns) - set(categorical_features) - set(float_features))
133        df[int_features] = df[int_features].astype(int)
134        # Handle numeric columns
135        for col in numerical_features:
136            df[col] = df[col].apply(lambda x: -100 if x < 0 else x)
137        # Handle categorical columns
138        for col in categorical_features:
139            df[col] = df[col].apply(
140                lambda x: (
141                    self.MISSING if x in cate_missing_values else str_utils.remove_vn_diacritics(x).lower().strip()
142                )
143            )
144        df.__is_norm = True
145        return df

147    def create_bin_data(self, df: pd.DataFrame, n_bins=10) -> dict[str, list[float | str]]:
148        """Compute bin thresholds for all features in *df*.
149
150        For **numeric** columns, generates ``n_bins`` percentile-based
151        edges from the positive (non-missing) values.  A leading ``0.0``
152        edge is always prepended so that the first bin captures the
153        zero/missing mass, and the final edge is nudged up by ``1e-10``
154        to make the last interval closed on the right.
155
156        For **categorical** columns (object dtype), collects the top
157        ``n_bins`` most frequent non-missing categories plus sentinel
158        values ``self.MISSING`` and ``self.OTHER``.
159
160        Parameters
161        ----------
162        df : pd.DataFrame
163            Normalised DataFrame (must have been processed by
164            ``normalize_data`` first).
165        n_bins : int, optional
166            Number of percentile bins for numeric features and maximum
167            number of categories to retain for categorical features.
168            Defaults to ``10``.
169
170        Returns
171        -------
172        pd.DataFrame
173            DataFrame with columns ``["feature_name", "type", "bins"]``:
174
175            - ``feature_name`` – Column name.
176            - ``type`` – Data type string (e.g. ``"float64"``, ``"int64"``,
177              ``"string"``).
178            - ``bins`` – List of numeric bin edges (float) for numeric
179              features, or list of category strings for categorical features.
180
181            Rows are sorted alphabetically by ``feature_name``.
182
183        Raises
184        ------
185        Exception
186            If *df* has not been normalised with ``normalize_data`` first.
187        """
188        if not self._check_norm(df):
189            raise Exception("DataFrame has not been normalized yet. Please use self.normalize_data(df)")
190        numerical_features = set(df.select_dtypes([int, float]).columns)
191        categorical_features = list(df.select_dtypes([object]).columns)
192        bin_thresholds = []
193        # Bin num features
194        for f in numerical_features:
195            series = df[f]
196            series = series[(series > 0) & (~series.isna())]
197            if len(series) == 0:
198                bins = np.array([])
199            else:
200                percentage = np.linspace(0, 100, n_bins + 1)
201                bins = np.percentile(series, percentage)
202                bins = [num_utils.round(e, series.dtype) for e in bins]
203                bins = np.unique([0.0, *bins])
204                if len(bins) >= 2:
205                    bins[-1] = bins[-1] + 1e-10
206            bin_thresholds.append([f, str(series.dtype).lower(), bins.tolist()])
207        # Bin cate features
208        for f in categorical_features:
209            series = df[f]
210            bins = series[series != self.MISSING].value_counts()[:n_bins].index.to_list()
211            bins = sorted(set([self.MISSING, self.OTHER, *bins]))
212            bin_thresholds.append([f, "string", bins])
213
214        df_bins = pd.DataFrame(bin_thresholds, columns=["feature_name", "type", "bins"])
215        return df_bins.sort_values("feature_name").reset_index(drop=True)

317    def calc_feature_distribution(self, df: pd.DataFrame, df_bins: dict[str, np.ndarray]) -> pd.DataFrame:
318        """Compute per-bin counts for each feature using the provided bin schema.
319
320        For **numeric** features, uses ``np.histogram`` with ``[-inf, *bins, +inf]``
321        boundaries so that out-of-range values are captured in the first
322        (``"missing"``) and last (``"other"``) segments.
323
324        For **categorical** features, maps any value not present in the
325        stored category list to ``self.OTHER`` and counts occurrences per
326        category.
327
328        Segment labels follow the pattern ``"A. [lo, hi)"`` (numeric) or
329        ``"A. category_name"`` (categorical), where the letter prefix
330        (A–Z) allows consistent ordering when displayed in BI tools.
331
332        Parameters
333        ----------
334        df : pd.DataFrame
335            Normalised DataFrame (must have been processed by
336            ``normalize_data`` first).
337        df_bins : pd.DataFrame
338            Bin-threshold table as returned by ``load_bin_data`` or
339            ``create_bin_data``.
340
341        Returns
342        -------
343        pd.DataFrame
344            Exploded distribution table with columns:
345
346            - ``feature_name`` – Feature column name.
347            - ``segment`` – Labelled bin string (e.g. ``"A. [0, 100)"``).
348            - ``count`` – Number of observations in this segment.
349            - ``total`` – Total observations for this feature.
350            - ``percent`` – Fraction of observations in this segment.
351
352        Raises
353        ------
354        Exception
355            If *df* has not been normalised with ``normalize_data`` first.
356        """
357        if not self._check_norm(df):
358            raise Exception("DataFrame has not been normalized yet. Please use self.normalize_data(df)")
359        bin_thresholds: dict[str, np.ndarray] = dict(zip(df_bins.feature_name, df_bins.bins, strict=True))
360        hists = []
361        categorical_features = [k for k, v in bin_thresholds.items() if len(v) > 0 and isinstance(v[0], str)]
362        numerical_features = [k for k, v in bin_thresholds.items() if len(v) > 0 and not isinstance(v[0], str)]
363
364        for f in numerical_features:
365            series = df[f]
366            hist, _ = np.histogram(series, [-np.inf, *bin_thresholds[f], np.inf])
367            segments = ["missing", *self._cvt_bins2labels(bin_thresholds[f]), "other"]
368            segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, segments, strict=False)]
369            hists.append([f, segments, hist, hist.sum(), hist / hist.sum()])
370        for f in categorical_features:
371            series = df[f]
372            bins = bin_thresholds[f]
373            series = series.apply(lambda x, b=bins: x if x in b else self.OTHER)
374            vc = series.value_counts()
375            for bin_name in bins:
376                if bin_name not in vc.index:
377                    vc.loc[bin_name] = 0
378            vc = vc.reindex(bins)
379            segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, bins, strict=False)]
380            hists.append([f, segments, vc, vc.sum(), vc / vc.sum()])
381
382        return pd.DataFrame(hists, columns=["feature_name", "segment", "count", "total", "percent"]).explode(
383            ["segment", "count", "percent"]
384        )

195            series = df[f]
196            series = series[(series > 0) & (~series.isna())]

199            else:

206            bin_thresholds.append([f, str(series.dtype).lower(), bins.tolist()])
207        # Bin cate features
208        for f in categorical_features:
209            series = df[f]
210            bins = series[series != self.MISSING].value_counts()[:n_bins].index.to_list()
211            bins = sorted(set([self.MISSING, self.OTHER, *bins]))
212            bin_thresholds.append([f, "string", bins])

318        """Compute per-bin counts for each feature using the provided bin schema.
319
320        For **numeric** features, uses ``np.histogram`` with ``[-inf, *bins, +inf]``
321        boundaries so that out-of-range values are captured in the first
322        (``"missing"``) and last (``"other"``) segments.
323
324        For **categorical** features, maps any value not present in the
325        stored category list to ``self.OTHER`` and counts occurrences per
326        category.
327
328        Segment labels follow the pattern ``"A. [lo, hi)"`` (numeric) or
329        ``"A. category_name"`` (categorical), where the letter prefix
330        (A–Z) allows consistent ordering when displayed in BI tools.
331
332        Parameters
333        ----------
334        df : pd.DataFrame
335            Normalised DataFrame (must have been processed by
336            ``normalize_data`` first).
337        df_bins : pd.DataFrame
338            Bin-threshold table as returned by ``load_bin_data`` or
339            ``create_bin_data``.
340
341        Returns
342        -------
343        pd.DataFrame
344            Exploded distribution table with columns:
345
346            - ``feature_name`` – Feature column name.
347            - ``segment`` – Labelled bin string (e.g. ``"A. [0, 100)"``).
348            - ``count`` – Number of observations in this segment.
349            - ``total`` – Total observations for this feature.
350            - ``percent`` – Fraction of observations in this segment.
351
352        Raises
353        ------
354        Exception
355            If *df* has not been normalised with ``normalize_data`` first.
356        """
357        if not self._check_norm(df):
358            raise Exception("DataFrame has not been normalized yet. Please use self.normalize_data(df)")
359        bin_thresholds: dict[str, np.ndarray] = dict(zip(df_bins.feature_name, df_bins.bins, strict=True))
360        hists = []
361        categorical_features = [k for k, v in bin_thresholds.items() if len(v) > 0 and isinstance(v[0], str)]
362        numerical_features = [k for k, v in bin_thresholds.items() if len(v) > 0 and not isinstance(v[0], str)]
363
364        for f in numerical_features:
365            series = df[f]
366            hist, _ = np.histogram(series, [-np.inf, *bin_thresholds[f], np.inf])
367            segments = ["missing", *self._cvt_bins2labels(bin_thresholds[f]), "other"]
368            segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, segments, strict=False)]
369            hists.append([f, segments, hist, hist.sum(), hist / hist.sum()])
370        for f in categorical_features:
371            series = df[f]
372            bins = bin_thresholds[f]
373            series = series.apply(lambda x, b=bins: x if x in b else self.OTHER)
374            vc = series.value_counts()
375            for bin_name in bins:
376                if bin_name not in vc.index:
377                    vc.loc[bin_name] = 0
378            vc = vc.reindex(bins)
379            segments = [". ".join(e) for e in zip(str_utils.UPPER_ALPHABET, bins, strict=False)]
380            hists.append([f, segments, vc, vc.sum(), vc / vc.sum()])
381
382        return pd.DataFrame(hists, columns=["feature_name", "segment", "count", "total", "percent"]).explode(
383            ["segment", "count", "percent"]
384        )
