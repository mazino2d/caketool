caketool.eda

EDA (Exploratory Data Analysis) module for caketool.

Supports pandas DataFrames with Plotly as the sole visualization backend.

Quick start

>>> import pandas as pd
>>> from caketool import eda
>>>
>>> df = pd.read_csv("data.csv")
>>>
>>> # Dataset overview
>>> eda.profile(df)
>>> eda.plot_correlations(df)
>>>
>>> # Univariate
>>> eda.plot_numeric_distribution(df["age"])
>>> eda.plot_categorical_frequency(df["category"])
>>>
>>> # Bivariate
>>> eda.plot_scatter(df, x="income", y="spend", color_by="segment")
>>> eda.plot_distribution_by_group(df, cat_col="segment", num_col="income", mode="box")
>>>

 97    """Compute a percentile summary table.
 98
 99    Parameters
100    ----------
101    series : pd.Series
102        Numeric series.
103    step : int
104        Percentile step size (e.g. 5 → 0%, 5%, 10%, ..., 100%).
105    low_trim : float
106        Lower quantile clip before computing.
107    high_trim : float
108        Upper quantile clip before computing.
109
110    Returns
111    -------
112    pd.DataFrame
113        Columns: percentile, value, count_below, count_above.
114    """
115    if not isinstance(step, int) or isinstance(step, bool) or not (1 <= step <= 100):
116        msg = "step must be an integer in [1, 100]"
117        raise ValueError(msg)
118
119    require_numeric(series)
120    require_nonempty(series)
121    s = clip_quantiles(series.dropna(), low_trim, high_trim)
122    value_col = f"{series.name}_value"
123
124    q1 = float(s.quantile(0.25))
125    q3 = float(s.quantile(0.75))
126    iqr = q3 - q1
127    stat_rows = [
128        {"percentile": "Mean", value_col: float(s.mean())},
129        {"percentile": "Median", value_col: float(s.median())},
130        {"percentile": "Std", value_col: float(s.std())},
131        {"percentile": "Min", value_col: float(s.min())},
132        {"percentile": "Max", value_col: float(s.max())},
133        {"percentile": "Q1", value_col: q1},
134        {"percentile": "Q3", value_col: q3},
135        {"percentile": "lower_fence", value_col: max(q1 - 1.5 * iqr, float(s.min()))},
136        {"percentile": "upper_fence", value_col: min(q3 + 1.5 * iqr, float(s.max()))},
137    ]
138
139    # Always include fine-grained extremes
140    pcts = sorted(set([0.01, 0.1, 0.5, 1.0] + list(range(0, 101, step)) + [99.0, 99.5, 99.9, 99.99]))
141    pcts = [p for p in pcts if 0 <= p <= 100]
142    percentile_rows = [{"percentile": f"{p:.2f}%", value_col: float(s.quantile(p / 100))} for p in pcts]
143
144    result = pd.DataFrame(stat_rows + percentile_rows)
145    result["count_below"] = result[value_col].map(lambda threshold: int((s <= threshold).sum()))
146    result["count_above"] = len(s) - result["count_below"]
147    return result.reset_index(drop=True)

274    """Compute value counts with frequency percentage.
275
276    Returns the *top_k* most frequent non-null values, an ``Others`` row
277    aggregating remaining categories (if any), and a ``NaN`` row when
278    ``dropna=False`` and missing values exist.
279
280    Parameters
281    ----------
282    series : pd.Series
283        Any series.
284    top_k : int
285        Limit to top *k* most frequent non-null values.
286    dropna : bool
287        If False, include a NaN row in the result when missing values exist.
288
289    Returns
290    -------
291    pd.DataFrame
292        Columns: value, count, pct, cumulative_count, cumulative_pct.
293    """
294    if not isinstance(top_k, int) or isinstance(top_k, bool) or top_k < 0:
295        msg = "top_k must be a non-negative integer"
296        raise ValueError(msg)
297
298    require_nonempty(series)
299    total = len(series)
300    nan_count = int(series.isna().sum())
301
302    # Non-null value counts
303    vc = series.value_counts(dropna=True)
304    top_vc = vc.head(top_k)
305    others_count = int(vc.iloc[top_k:].sum()) if len(vc) > top_k else 0
306
307    rows: list[dict] = [{"value": val, "count": cnt} for val, cnt in zip(top_vc.index, top_vc.values, strict=True)]
308    if others_count > 0:
309        rows.append({"value": "Others", "count": others_count})
310    if not dropna and nan_count > 0:
311        rows.append({"value": "NaN", "count": nan_count})
312
313    df = pd.DataFrame(rows, columns=["value", "count"])
314    df["pct"] = (df["count"] / total * 100).round(2)
315    df["cumulative_count"] = df["count"].cumsum()
316    df["cumulative_pct"] = (df["cumulative_count"] / total * 100).round(2)
317    return df.reset_index(drop=True)
